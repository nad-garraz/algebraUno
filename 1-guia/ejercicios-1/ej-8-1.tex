\begin{enunciado}{\ejercicio}
  Hallar el conjunto $\partes(A)$ de partes de $A$ en los casos.
  \begin{multicols}{3}
    \begin{enumerate}[label=\roman*)]
      \item $A = \set{1}$
      \item $A = \set{a,b}$
      \item $A = \set{1,\set{1,2},3}$
    \end{enumerate}
  \end{multicols}
\end{enunciado}

  Recordando que:\par
$
  \partes(A) \to
  \begin{cases}
    B \en \partes(A) \sisolosi B \subseteq A \\
    \partes(A) = \set{B \talque B \subseteq A}
  \end{cases}
$
\begin{enumerate}[label=(\roman*)]
  \item $A = \set{1} \to \partes(A) = \set{\vacio, A}$ \Tilde
  \item $A =
          \set{a, b}
          \to
          \partes(A) =
          \set{
            \vacio,
            \set{a},
            \set{b},
            A
          }$\Tilde
  \item $
          A =
          \set{1, \set{1,2}, 3}
          \to
          \partes(A) =
          \set{
            \vacio,
            \set{1},
            \set{\set{1,2}},
            \set{3},
            \set{1,\set{1,2}},
            \set{1,3},
            \set{\set{1,2},3},
            A
          }$\Tilde
\end{enumerate}
