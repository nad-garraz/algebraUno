\begin{enunciado}{\ejercicio}
  Proba que las siguientes afirmaciones son verdaderas para todo $n \en \naturales.$
  \begin{multicols}{2}
    \begin{enumerate}[label=\alph*)]
      \item $99 \divideA 10^{2n} + 197$
      \item $9 \divideA 7 \cdot 5^{2n} + 2^{4n+1}$
      \item $56 \divideA 13^{2n} + 28n^2 - 84n -1$
      \item $256 \divideA 7^{2n} + 208n - 1$
    \end{enumerate}
  \end{multicols}
\end{enunciado}

\parrafoDestacado[\atencion]{
  \it
  Por ser el primer ejercicio de este estilo algunos pasos, son evitables/redundantes, pero siguen siendo válidos.
  Más adelante los ejercicios van a estar hechos usando resultados de forma más directa.
}

\begin{enumerate}[label=\alph*)]
  \item
        $$
          \begin{array}{rcl}
            99 \divideA 10^{2n} + 197 & \Sii{def}      & \congruencia{10^{2n} + 197}{0}{99}                    \\
                                      & \sii           & \congruencia{10^{2n} + 198}{1}{99}                    \\
                                      & \Sii{\red{!}}  & \congruencia{10^{2n} + \ub{198}{\conga{99} 0}}{1}{99} \\
                                      & \sii           & \congruencia{100^n}{1}{99}                            \\
                                      & \Sii{\red{!!}} & \congruencia{1^n}{1}{99}                              \\
                                      & \Sii{def}      & 99 \divideA 1 - 1 \sii 99 \divideA 0                  \\
          \end{array}
        $$
        En el último \red{!!} uso que $a^n \conga{b} (r_b(a))^n$ que funciona por algo así (\textit{pseudo-demo}):
        $$
          \begin{array}{rcl}
            \congruencia{100}{1}{99} & \Sii{$\times 100$}[M.A.M] & \congruencia{100^2}{\ub{100}{ \conga{99} 1 }}{99} \sii \congruencia{100^2}{1}{99} \\\\
                                     & \Sii{$\times 100$}[M.A.M] & \congruencia{100^3}{100}{99} \sii \congruencia{100^3}{1}{99}                      \\
                                     & \vdots                    & \vdots                                                                            \\
                                     & \Sii{$\times 100$}[M.A.M] & \congruencia{100^n}{100}{99} \sii \cajaResultado{\congruencia{100^n}{1}{99} }     \\
          \end{array}
        $$

        Se concluye que:
        $$
          \cajaResultado{
            99 \divideA 10^{2n} + 197
          }
        $$

  \item Parecido al anterior:
        $$
          \begin{array}{rcl}
            9 \divideA 7 \cdot 5^{2n} + 2^{4n+1}
             & \Sii{def}                         & \congruencia{7\cdot5^{2n} + 2^{4n+1}}{0}{9}                                     \\
             & \Sii{sumo $2\cdot 5^{2n}$}[M.A.M] & \congruencia{\ub{9 \cdot 5^{2n}}{\conga9 0} + 2\cdot 2^{4n}}{2 \cdot 5^{2n}}{9} \\
             & \sii                              & \congruencia{2^{4n}}{5^{2n}}{9}                                                 \\
             & \sii                              & \congruencia{16^n}{25^n}{9}                                                     \\
             & \Sii{\red{!}}                     & \congruencia{7^n}{7^n}{9}                                                       \\
             & \Sii{def}                         & 9 \divideA 0
          \end{array}
        $$
        Se concluye que:
        $$
          \cajaResultado{
            9 \divideA 7 \cdot 5^{2n} + 2^{4n+1}
          }
        $$

  \item $56 \divideA 13^{2n} + 28n^2 - 84n -1
      \Sii{def}
        \congruencia{13^{2n} + 28n^2 - 84n -1}{0}{56}$.

        Lo \textit{hagaimos} por inducción:
        $$
          p(n) : \congruencia{13^{2n} + 28n^2 - 84n -1}{0}{56}
        $$

        \textit{Caso base:}
        $$
          p(1) : \congruencia{\ub{13^2 + 28 - 84 - 1}{\conga{56} 0}}{0}{56}
        $$
        El caso base resultó \textit{verdadero}.

        \medskip
        \textit{Paso inductivo:}

        Asumimos $p(n)$ verdadera, queremos ver que $p(n + 1)$ también lo sea.
        $$
          \begin{array}{rcl}
            \congruencia{13^{2n + 2} + 28(n+1)^2 - 84(n+1) - 1}{0}{56}
             & \sii              & \congruencia{13^{2n} \cdot 13^2 + 28n^2 + 56n + 28 -84n - 84 - 1}{0}{56}                                                \\
             & \Sii{\purple{HI}} & \congruencia{\ub{13^{2n} + 28n^2 + 56n -84n}{\purple{\text{hipótesis inductiva}}} + 28 - 84 + 13^2 - 1}{0}{56} \\
             & \sii              & \congruencia{28 - 84 + 168}{0}{56}                                                                                      \\
             & \sii              & \congruencia{112}{0}{56}
          \end{array}
        $$

        Se probó que $p(n + 1)$ es verdadera, luego $p(n)$ es verdadera para todo $n \en \naturales$.

  \item
        Hermoso ejercicio en el que sin fe, en el \textit{todo poderoso Gauss}, sencillamente uno tira la toalla antes de tiempo.

        Sale por inducción:

        \textit{Quiero ver que la siguiente proposición es verdadera:}
        $$
          p(n) : 256 \divideA 49^n + 208n - 1
        $$

        O en notación de congruencia:

        $$
          p(n) : \congruencia{49^n + 208n - 1}{0}{256}
        $$

        \textit{Caso base:}
        $$
          p(\blue{1}) : 256 \divideA 49^{\blue{1}} + 208 \cdot \blue{1} - 1 \Tilde
        $$
        Por lo tanto $p(1)$ resulta verdadera.

        \textit{Paso inductivo:}

        Uso la notación de congruencia de acá en adelante, porque es mucho más cómodo.
        Supongo que:
        $$
          p(\blue{k}) : \ub{\congruencia{49^{\blue{k}} + 208  \blue{k} - 1}{0}{256}}{\purple{\text{hipótesis inductiva}}}
          \quad \text{ para algún } \blue{k} \en \enteros
        $$
        es  una proposición verdadera.
        Entonces quiere probar que:
        $$
          p(\blue{k+1}) : \congruencia{49^{\blue{k+1}} + 208  (\blue{k+1}) - 1}{0}{256},
        $$
        también sea verdadera. Arranco del paso $(k+1)$ y haciendo un poco de \textit{matemagia}:
        $$
          \begin{array}{rcl}
            49^{k+1} + 208 \cdot (k+1) - 1 & =                       & 49 \cdot 49^k + 208k + 208 - 1 \\
              &\conga{256}[\purple{\text{HI}}]& 49 \cdot (\purple{-208k + 1}) + 208k + 208 - 1 \\\\
                                           & \conga{256}             & 49 \cdot (48k + 1) - 48k -48 - 1 = 2352 k + 49 - 48k - 49 \\
                                           & \conga{256}[\red{!!}] & 48 k + 49 - 48k - 49 = 0 \Tilde
          \end{array}
        $$
        En \red{!!} y gracias a Gauss $\congruencia{2352}{48}{256}$ ¿Casualidad? No sé y no me importa.\par

        Dado que $\congruencia{49^{k+1} + 208 \cdot (k+1) - 1}{0}{256}$, la proposición $p(k+1)$ resultó verdadera.\par\bigskip

        Dado que $p(1), p(k) \ytext p(k+1)$ resultaron verdaderas, por principio de inducción $p(n)$ también lo es para todo $n \en \naturales$.

\end{enumerate}
