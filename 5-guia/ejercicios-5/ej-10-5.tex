\begin{enunciado}{\ejercicio}
  Hallar, cuando existan, todos los enteros $a$ que satisfacen simultáneamente:

  \begin{enumerate}[label=\roman*)]
    \begin{multicols}{3}
      \item
      $
        \llave{ll}{
          \congruencia{a}{3}{10} \\
          \congruencia{a}{2}{7}  \\
          \congruencia{a}{5}{9}
        }
      $
      \item
      $
        \llave{ll}{
          \congruencia{a}{1}{6}  \\
          \congruencia{a}{2}{20} \\
          \congruencia{a}{3}{9}
        }
      $

      \item $
        \llave{l}{
          \congruencia{a}{1}{12} \\
          \congruencia{a}{7}{10} \\
          \congruencia{a}{4}{9}
        }
      $
    \end{multicols}
  \end{enumerate}

\end{enunciado}

\begin{enumerate}[label=\roman*)]
  \item Hay que resolver el sistema de ecuaciones de congruencia. Tengo
        divisores coprimos 2 a 2, así que por el \href{\chinito}{teorema
          chino del resto} hay solución:
        $$
          \llave{ll}{
            \congruencia{a}{3}{10} & \llamada1 \\
            \congruencia{a}{2}{7}  & \llamada2 \\
            \congruencia{a}{5}{9}  & \llamada3 \\
          }
        $$
        El sistema tiene solución dado que 10, 7 y 9 son coprimos dos a dos. Resuelvo empezando por $\llamada1$ despejando
        y reemplazando en las demás ecuaciones:
        $$
          \congruencia{a}{3}{10}
          \Sii{def}
          a = 10\blue{k} + 3 \conga{7}
        $$
        Reemplazo ahora en $\llamada2$:
        $$
          \congruencia{10 \blue{k} +3}{2}{7}
          \sii
          \congruencia{3\blue{k}}{6}{7}
          \Sii{$3 \cop 7$}
          \congruencia{\blue{k}}{2}{7}
          \Sii{def}
          \blue{k} = 7\yellow{j} + 2
        $$
        Reemplazo el valor de $\blue{k}$ en $a$:
        $$
          a = 10 \cdot ( 7\yellow{j} + 2) + 3 = 70 \yellow{j} + 23
        $$
        Y ahora reemplazo el valor de $a$ en $\llamada3$:
        $$
          \congruencia{70 \yellow{j} + 23}{5}{9}
          \sii
          \congruencia{7 \yellow{j}}{0}{9}
          \Sii{$7 \cop 9$}
          \congruencia{\yellow{j}}{0}{9}
          \Sii{def}
          \yellow{j} = 9\purple{h}
        $$
        \textit{Máquina de hacer chorizos}
        y ahora reemplazo el valor de $\yellow{j}$ en $a$:
        $$
          a = 70(9\purple{h}) + 23 = 630 \purple{h} + 23
          \Sii{def}
          \congruencia{a}{23}{630}
        $$
        El TCH nos \textit{aseguraba} una solución en el
        intervalo $[0, 630)$ \Tilde

  \item Quebrando se ve que es \textit{incompatible}. Los sistemas \textit{quebrados} tienen que
        contener las soluciones del sistema \textit{original}:
        $$
          \llave{l}{
            \congruencia{a}{1}{6}
            \equivalente
            \llave{l}{
              \congruencia{a}{1}{2}~~ \red{\skull}  \\
              \congruencia{a}{1}{3}~~ \blue{\skull}
            }\\
            \congruencia{a}{2}{20}
            \equivalente
            \llave{l}{
              \congruencia{a}{1}{5}\\
              \congruencia{a}{0}{2} ~~ \red{\skull}
            }\\
            \congruencia{a}{3}{9}
            \equivalente
            \llave{l}{
              \congruencia{a}{0}{3}~~ \blue{\skull}
            }
          }
        $$
        Lo cual no puede ocurrir, dado que \red{por un lado} pido que $a$ sea par y por otro impar. \blue{Por otro lado} pido algo similar
        con los múltiplos de 3.

  \item Quebrando se ve que es \textit{compatible}.
        $$
          \llave{rcl}{
            \congruencia{a}{1}{12}
            &\equivalente &
            \llave{l}{
              \congruencia{a}{1}{4}
              \equivalente
              \llave{l}{
                \congruencia{a}{1}{2}
              }
              \\
              \congruencia{a}{1}{3}
            }\\
            \congruencia{a}{7}{10}
            &\equivalente &
            \llave{l}{
              \congruencia{a}{1}{2}  \\
              \congruencia{a}{2}{5}
            }\\
            \congruencia{a}{4}{9}
            &\equivalente &
            \llave{l}{
              \congruencia{a}{1}{3}
            }
          }
        $$
        Todos los sistemas  \textit{quebrados} van a tener soluciones dentro de las soluciones de sus \textit{antecesores}. Los sistemas que uso
        para resolver son los que tiene los \textit{mayores divisores}, \magenta{siempre de manera que los divisores queden coprimos 2 a 2}:
        $$
          \llave{l}{
            \congruencia{a}{1}{4}\\
            \congruencia{a}{2}{5}\\
            \congruencia{a}{4}{9}
          }
        $$
        Por \href{\chinito}{TCH} esto tiene solución. Eso queda para que lo hagas vos, \textit{because pajilla}. Si no está mal sería:
        $$
          \congruencia{a}{157}{180}
        $$
\end{enumerate}
