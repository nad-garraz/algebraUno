\begin{enunciado}{\ejercicio}
    Hallar todos los $a\en \enteros$ para los cuales $(2a-3:4a^{2}+10a-10) \neq 1$ 
   \end{enunciado}

   $$
     \llave{l}{
       d \divideA 2a-3 \Entonces{.2a}[] d \divideA 4a^{2}-6 \\

     }
     \llave{l}{
       d \divideA 4a^{2}-6 \\
       d \divideA 4a^{2}+10a-10
     }
   \Entonces{Resto}[]
   d \divideA 16a-10
   $$

   $$
   \llave{l}{

   $$
